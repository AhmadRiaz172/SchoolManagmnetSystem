select products.name,
  cast(stock.price/stock.noofitems as decimal(10,2)) as PurchasePrice,
    products.price as SellingPrice,
      (products.price-cast(stock.price/stock.noofitems as decimal(10,2)))   as Profit
from products
inner join stock on products.productid = stock.productid;



select sum(orders.price - cast((stock.price/stock.noofitems) as decimal(10,2))*orders.orderquantity - shipper.deliverycharges ) as "Totall Profit"
from orders
inner join stock on stock.productid = orders.productid
inner join shipper on shipper.shipper_id = orders.shipperid;
