                                     
 CREATE TABLE "SCOTT"."CUSTOMER" 
   ( "CUSTOMER_ID" VARCHAR2(10 BYTE), 
	 "CONTACT_NO" VARCHAR2(11 BYTE), 
	 "NAME" VARCHAR2(20 BYTE), 
	 CONSTRAINT "CUSTOMER_CHK1" CHECK (CONTACT_NO NOT LIKE '%[^0-9]%')  
   )
   ;
   
   CREATE TABLE "SCOTT"."SHIPPER"
(
  SHIPPER_ID VARCHAR2(5) NOT NULL,
  CONTACT_NO VARCHAR2(11),
  ADDRESS VARCHAR2(100)
, CONSTRAINT SHIPPER_PK PRIMARY KEY
  (
    SHIPPER_ID
  )
  ENABLE
)
;


  CREATE TABLE "SCOTT"."SUPPIER" 
   (	"SUPPLIER_ID" VARCHAR2(4 BYTE) NOT NULL ENABLE, 
	"NAME" VARCHAR2(20 BYTE), 
	"CONTACTNO" VARCHAR2(11 BYTE), 
	 CONSTRAINT "SUPPIER_PK" PRIMARY KEY ("SUPPLIER_ID")
  USING INDEX PCTFREE 10 INITRANS 2 MAXTRANS 255 NOCOMPRESS LOGGING
  TABLESPACE "USERS"  ENABLE, 
	 CONSTRAINT "SUPPIER_PHONECHK" CHECK (CONTACTNO NOT LIKE '%[^0-9]%') ENABLE
   )
   ;
   
   
   CREATE TABLE "SCOTT"."CATEGORY" 
   (	"NAME" VARCHAR2(20 BYTE), 
	"ID" VARCHAR2(5 BYTE) NOT NULL ENABLE, 
	 CONSTRAINT "CATEGORY_PK" PRIMARY KEY ("ID")
  USING INDEX PCTFREE 10 INITRANS 2 MAXTRANS 255 NOCOMPRESS LOGGING
  TABLESPACE "USERS"  ENABLE
   )
   ;
   
   

CREATE TABLE "SCOTT"."STOCK" 
   (	"STOCK_ID" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"PRICE" NUMBER(10,0) NOT NULL ENABLE, 
	"DATE" DATE, 
	"NOOFITEMS" NUMBER(6,0), 
	"SUPPLIERID" VARCHAR2(5 BYTE), 
	"PRODUCTID" VARCHAR2(5 BYTE), 
	 CONSTRAINT "STOCK_PK" PRIMARY KEY ("STOCK_ID")
  USING INDEX PCTFREE 10 INITRANS 2 MAXTRANS 255 COMPUTE STATISTICS 
  STORAGE(INITIAL 65536 NEXT 1048576 MINEXTENTS 1 MAXEXTENTS 2147483645
  PCTINCREASE 0 FREELISTS 1 FREELIST GROUPS 1 BUFFER_POOL DEFAULT FLASH_CACHE DEFAULT CELL_FLASH_CACHE DEFAULT)
  TABLESPACE "USERS"  ENABLE, 
	 CONSTRAINT "STOCK_PRODUCTS_FK1" FOREIGN KEY ("PRODUCTID")
	  REFERENCES "SCOTT"."PRODUCTS" ("PRODUCTID") ENABLE
   )
   ;


CREATE TABLE "SCOTT"."PRODUCTS" 
   (	"PRODUCTID" VARCHAR2(4 BYTE) NOT NULL ENABLE, 
	"NAME" VARCHAR2(20 BYTE) NOT NULL ENABLE, 
	"EXPIRYDATE" DATE, 
	"PRICE" NUMBER(10,0), 
	"SUPPLIERID" VARCHAR2(5 BYTE), 
	"CATEGORYID" VARCHAR2(5 BYTE), 
	 CONSTRAINT "PRODUCTS_PK" PRIMARY KEY ("PRODUCTID")
  USING INDEX PCTFREE 10 INITRANS 2 MAXTRANS 255 COMPUTE STATISTICS 
  STORAGE(INITIAL 65536 NEXT 1048576 MINEXTENTS 1 MAXEXTENTS 2147483645
  PCTINCREASE 0 FREELISTS 1 FREELIST GROUPS 1 BUFFER_POOL DEFAULT FLASH_CACHE DEFAULT CELL_FLASH_CACHE DEFAULT)
  TABLESPACE "USERS"  ENABLE
   )
   ;
   

  CREATE TABLE "SCOTT"."ORDERS" 
   (	"CUSTOMERID" VARCHAR2(5 BYTE), 
	"PRODUCTID" VARCHAR2(5 BYTE), 
	"ORDERQUANTITY" NUMBER(10,0), 
	"PRICE" NUMBER(20,0), 
	"EMPLOYEEID" VARCHAR2(5 BYTE), 
	"SHIPPERID" VARCHAR2(5 BYTE), 
	 CONSTRAINT "ORDERS_PRODUCTS_FK1" FOREIGN KEY ("PRODUCTID")
	  REFERENCES "SCOTT"."PRODUCTS" ("PRODUCTID") ENABLE
   );
   
   
   
   
CREATE TABLE "SCOTT"."EMPLOYEE"
